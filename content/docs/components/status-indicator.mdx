## Installation

```bash
npx shadcn@latest add @shoscn/status-indicator
```

## Usage

```tsx
import { StatusIndicator } from '@/components/ui/utils/status-indicator';

export default function MyComponent() {
  return <StatusIndicator variant="success" />;
}
```

## API Reference

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `variant` | `primary \| secondary \| muted \| accent \| success \| warning \| destructive` | - | The color variant of the indicator |
| `className` | `string` | - | Additional CSS classes |
