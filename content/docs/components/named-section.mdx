## Installation

```bash
npx shadcn@latest add @shoscn/named-section
```

## Usage

```tsx
import { NamedSection } from '@/components/ui/layouts/named-section';

export default function MyComponent() {
  return (
    <NamedSection title="Section Title">
      <p>Section content goes here</p>
    </NamedSection>
  );
}
```

## API Reference

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `title` | `string` | - | The section title displayed at the top |
| `menu` | `React.ReactNode` | - | Optional menu or actions displayed on the right side |
| `children` | `React.ReactNode` | - | Content to display inside the section |
