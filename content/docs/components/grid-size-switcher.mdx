## Usage

```tsx
import { useState } from 'react';
import { GridSizeSwitcher } from '@/components/ui/utils/grid-size-switcher';
import { Grid, GridSize } from '@/components/ui/layouts/grid';

export default function MyComponent() {
  const [size, setSize] = useState<GridSize>('medium');

  return (
    <div>
      <GridSizeSwitcher value={size} onChange={setSize} />
      <Grid size={size}>
        {/* Grid items */}
      </Grid>
    </div>
  );
}
```

## API Reference

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `value` | `GridSize` | - | Current grid size value |
| `onChange` | `(size: GridSize) => void` | - | Callback when grid size changes |
| `buttonSize` | `'default' \| 'sm' \| 'lg' \| 'icon'` | `'default'` | Size of the trigger button |
| `align` | `'start' \| 'center' \| 'end'` | `'end'` | Alignment of the popover content |

## GridSize Type

The `GridSize` type supports the following values:

- `'tiny'` - Extra small grid with many columns
- `'small'` - Small grid with moderate columns
- `'medium'` - Medium-sized grid (default)
- `'large'` - Large grid with fewer columns
