## Installation

```bash
npx shadcn@latest add @shoscn/vertical-box
```

## Usage

```tsx
import { VerticalBox } from '@/components/ui/layouts/vertical-box';

export default function MyComponent() {
  return (
    <VerticalBox>
      <div>Top</div>
      <div>Bottom</div>
    </VerticalBox>
  );
}
```

## API Reference

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `children` | `React.ReactNode` | - | Child elements to display in the vertical layout |
| `reversed` | `boolean` | `false` | Reverse the order of children (bottom to top) |
