## Installation

```bash
npx shadcn@latest add @shoscn/flex
```

## Usage

```tsx
import { Flex } from '@/components/ui/layouts/flex';

export default function MyComponent() {
  return (
    <Flex direction="row" justify="center" align="center">
      <div>Item 1</div>
      <div>Item 2</div>
      <div>Item 3</div>
    </Flex>
  );
}
```

## API Reference

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `children` | `React.ReactNode` | - | Child elements to display in the flex layout |
| `direction` | `'row' \| 'row-reverse' \| 'col' \| 'col-reverse'` | `'row'` | Flex direction |
| `justify` | `'start' \| 'end' \| 'center' \| 'between' \| 'around' \| 'evenly'` | `'start'` | Justify content along the main axis |
| `align` | `'start' \| 'end' \| 'center' \| 'baseline' \| 'stretch'` | `'start'` | Align items along the cross axis |
| `gap` | `string` | `'gap-4'` | Gap between flex items (Tailwind CSS class) |
